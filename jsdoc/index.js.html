/**
 * calculate area of rectangle
 * @param {number} value1 the length og rectangle
 * @param {number} value2 the width of rectangle
 * @returns {number} area of rectangle
 */
function area_of_rectangle(value1, value2) {
    return value1 * value2;
}


/**
 * calculate area of triangle
 * @param {number} value1 the base og triangle
 * @param {number} value2 the height of triangle
 * @returns {number} area of triangle
 */
function area_of_triangle(value1, value2) {
  return value1 * value2*0.5;
}



/**
 * calculate area of circle
 * @param {number} value1 the redies circle
 * @returns {number} area of circle
 */
function area_of_circle(value1) {
    return Math.PI * value1;
}
